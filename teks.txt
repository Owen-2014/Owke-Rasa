  <script>
    // App State
    const menus = {
      tiramisu: {
        title: "Tiramisu",
        sizes: {
          kecil: 25000,
          besar: 35000
        },
        svg: `<svg viewBox="0 0 100 100" class="w-full h-full"><path d="M20 70 L20 50 Q20 45 25 45 L75 45 Q80 45 80 50 L80 70 Q80 75 75 75 L25 75 Q20 75 20 70 Z" fill="#D7CCC8" /><rect x="20" y="52" width="60" height="6" fill="#8D6E63" /><rect x="20" y="62" width="60" height="6" fill="#5D4037" /><path d="M25 45 C30 40, 35 45, 40 42 C45 40, 50 45, 55 42 C60 40, 65 45, 70 42 C75 40, 80 45, 80 45" fill="none" stroke="#F5F5F5" stroke-width="6" stroke-linecap="round" /><circle cx="35" cy="40" r="4" fill="#E0F7FA" /><circle cx="50" cy="38" r="4" fill="#FF8A80" /><circle cx="65" cy="40" r="4" fill="#E8F5E9" /></svg>`
      },
      kue_bolu: {
        title: "Kue Bolu",
        sizes: {
          standar: 60000
        },
        svg: `<svg viewBox="0 0 100 100" class="w-full h-full"><ellipse cx="50" cy="65" rx="32" ry="16" fill="#F4B400" /><rect x="18" y="45" width="64" height="20" fill="#F4B400" /><ellipse cx="50" cy="45" rx="32" ry="16" fill="#FFE082" /><path d="M30 45 Q50 35 70 45" fill="none" stroke="#FFF9C4" stroke-width="4" stroke-linecap="round" /><circle cx="50" cy="42" r="5" fill="#D32F2F" /></svg>`
      },
      cheesecake: {
        title: "Cheesecake",
        sizes: {
          mini: 8000,
          kecil: 25000,
          besar: 35000
        },
        svg: `<svg viewBox="0 0 100 100" class="w-full h-full"><path d="M15 65 L80 75 L80 45 L15 35 Z" fill="#FFF59D" /><path d="M15 65 L80 75 L80 80 L15 70 Z" fill="#8D6E63" /><path d="M15 35 L80 45 L80 48 L15 38 Z" fill="#E57373" /></svg>`
      }
    };

    let currentSelection = {
      menuKey: null,
      size: null,
      qty: 1
    };

    const BIAYA_TAMBAHAN = 3000;
    let timerInterval = null;
    let toastTimeout1 = null;
    let toastTimeout2 = null;

    // Custom Alert Notification Function
    function showAlert(msg) {
      const alertEl = document.getElementById('custom-alert');
      const alertMsg = document.getElementById('custom-alert-msg');
      alertMsg.textContent = msg;
      
      alertEl.classList.remove('translate-y-20', 'opacity-0', 'pointer-events-none');
      alertEl.classList.add('translate-y-0', 'opacity-100');
      
      setTimeout(() => {
        alertEl.classList.remove('translate-y-0', 'opacity-100');
        alertEl.classList.add('translate-y-20', 'opacity-0', 'pointer-events-none');
      }, 4000);
    }

    // Navigation: Select Menu
    function selectMenu(menuKey) {
      currentSelection.menuKey = menuKey;
      currentSelection.size = null; // reset size selection
      currentSelection.qty = 1; // reset qty
      
      // Load selected data
      const menuData = menus[menuKey];
      document.getElementById('selected-menu-title').textContent = menuData.title;
      document.getElementById('selected-menu-svg').innerHTML = menuData.svg;
      document.getElementById('quantity-display').textContent = currentSelection.qty;
      
      // Render sizes options dynamically
      const sizeContainer = document.getElementById('size-options-container');
      sizeContainer.innerHTML = '';
      
      const sizeKeys = Object.keys(menuData.sizes);
      
      sizeKeys.forEach(size => {
        const price = menuData.sizes[size];
        const btn = document.createElement('button');
        btn.type = 'button';
        btn.id = `size-btn-${size}`;
        btn.className = "border-2 border-amber-300 text-[#5D4037] font-semibold py-3 px-2 rounded-2xl transition focus:outline-none hover:bg-amber-100 flex flex-col items-center justify-center active:scale-95";
        btn.onclick = () => selectSize(size);
        
        btn.innerHTML = `
          <span class="capitalize text-sm font-bold">${size}</span>
          <span class="text-xs text-amber-900 mt-1">Rp ${price.toLocaleString('id-ID')}</span>
        `;
        sizeContainer.appendChild(btn);
      });

      // Show purchasing screen
      switchScreen('screen-home', 'screen-purchase');
      updatePrice();
      checkFormValidity();
    }

    // Select Size & Update Visual
    function selectSize(size) {
      currentSelection.size = size;
      
      // Highlight selection
      const menuData = menus[currentSelection.menuKey];
      Object.keys(menuData.sizes).forEach(s => {
        const btn = document.getElementById(`size-btn-${s}`);
        if (btn) {
          if (s === size) {
            btn.classList.remove('border-amber-300', 'bg-transparent');
            btn.classList.add('border-[#A1887F]', 'bg-[#A1887F]/20');
          } else {
            btn.classList.remove('border-[#A1887F]', 'bg-[#A1887F]/20');
            btn.classList.add('border-amber-300', 'bg-transparent');
          }
        }
      });
      
      updatePrice();
      checkFormValidity();
    }

    // Quantity Counter +/-
    function updateQty(change) {
      let newQty = currentSelection.qty + change;
      if (newQty < 1) newQty = 1;
      currentSelection.qty = newQty;
      document.getElementById('quantity-display').textContent = newQty;
      updatePrice();
      checkFormValidity();
    }

    // Price calculator preview
    function updatePrice() {
      if (!currentSelection.menuKey) return;
      const menuData = menus[currentSelection.menuKey];
      let basePrice = 0;
      
      if (currentSelection.size && menuData.sizes[currentSelection.size]) {
        basePrice = menuData.sizes[currentSelection.size];
      }
      
      const totalPrice = (basePrice * currentSelection.qty) + BIAYA_TAMBAHAN;
      
      if (currentSelection.size) {
        document.getElementById('price-preview').textContent = `Rp ${totalPrice.toLocaleString('id-ID')}`;
      } else {

      const btn = document.getElementById('btn-payment-submit');
      if (currentSelection.menuKey && currentSelection.size && currentSelection.qty > 0) {

    // Generate Invoice Receipt logic
    function generateReceipt() {
      // Create Unique Receipt Code
      const randNum = Math.floor(1000 + Math.random() * 9000);
      const randId = `OWKE-RASA-${randNum}`;
      
      const menuData = menus[currentSelection.menuKey];
      const basePrice = menuData.sizes[currentSelection.size];
      const totalAmount = (basePrice * currentSelection.qty) + BIAYA_TAMBAHAN;

      // Populate Receipt Card Elements
      document.getElementById('receipt-id').textContent = `ID: ${randId}`;
      document.getElementById('receipt-item-name').textContent = menuData.title;
      document.getElementById('receipt-item-size').textContent = currentSelection.size.toUpperCase();
      document.getElementById('receipt-item-qty').textContent = `${currentSelection.qty}x`;
      document.getElementById('receipt-item-price').textContent = `Rp ${basePrice.toLocaleString('id-ID')}`;
      document.getElementById('receipt-total').textContent = `Rp ${totalAmount.toLocaleString('id-ID')}`;

      const receiptCard = document.getElementById('receipt-container');
      receiptCard.classList.remove('hidden');
      
      // Hide Save Receipt Button

    // Floating notifications sequence (Permanently displaying the copy button & number)
    function showFloatingNotifications() {
      const toast = document.getElementById('floating-toast');
      const toastText = document.getElementById('floating-toast-text');
      const copyBtn = document.getElementById('btn-copy-wa');
      const waDisplay = document.getElementById('wa-display-area');
      
      // Show Updated First Message
      toastText.textContent = "Screenshot Struk ini lalu kirim ke nomor wa dibawah";
      copyBtn.classList.remove('hidden');
      waDisplay.classList.remove('hidden');
      
      // Reset opacity
      toastText.classList.remove('opacity-0');
      copyBtn.classList.remove('opacity-0');
      waDisplay.classList.remove('opacity-0');

      toast.classList.remove('hidden');
      // Force trigger DOM reflow for animations
      void toast.offsetWidth;
      toast.classList.remove('scale-95', 'opacity-0');
      toast.classList.add('scale-100', 'opacity-100');

      // After 10 seconds, only transition the main instruction text.
      // The WhatsApp number (waDisplay) and Copy button (copyBtn) STAY completely visible.
      toastTimeout1 = setTimeout(() => {
        toastText.classList.add('opacity-0');
        
        toastTimeout2 = setTimeout(() => {
          toastText.textContent = "Kirim bukti pembayaran ke nomor tersebut untuk Pembayaran Final yaa 😄❤️";
          toastText.classList.remove('opacity-0');
        }, 300);
        
      }, 10000);

    // Clipboard Copy logic
    function copyWA() {
      const numberToCopy = "085694721512";
      
      const textArea = document.createElement("textarea");
      textArea.value = numberToCopy;
      textArea.style.position = "fixed";  // Avoid scrolling to bottom
      document.body.appendChild(textArea);
      textArea.focus();
      textArea.select();
      
      try {
        const successful = document.execCommand('copy');
        if (successful) {
          showAlert("Nomor WhatsApp berhasil disalin! 😊");
        } else {
          showAlert("Gagal menyalin nomor.");
        }
      } catch (err) {
        showAlert("Maaf, tidak dapat menyalin.");
      }
      
      document.body.removeChild(textArea);
    }

    // Countdown 3 Minutes implementation
    function startCountdown(durationSeconds) {
      clearInterval(timerInterval);
      let timer = durationSeconds;
      const display = document.getElementById('countdown-timer');

      const updateTimerDisplay = () => {
        let minutes = parseInt(timer / 60, 10);
        let seconds = parseInt(timer % 60, 10);

        minutes = minutes < 10 ? "0" + minutes : minutes;
        seconds = seconds < 10 ? "0" + seconds : seconds;

        display.textContent = minutes + ":" + seconds;

        if (--timer < 0) {
          clearInterval(timerInterval);
          showAlert("Batas waktu pembayaran habis! Transaksi Anda dibatalkan.");
          resetAndBackToHome();
        }
      };

      updateTimerDisplay(); // Initial trigger
      timerInterval = setInterval(updateTimerDisplay, 1000);
    }

    // Screen swapper transition helper
    function switchScreen(fromId, toId) {
      const fromEl = document.getElementById(fromId);
      const toEl = document.getElementById(toId);
      
      fromEl.classList.add('opacity-0', 'transition-opacity', 'duration-200');
      setTimeout(() => {
        fromEl.classList.add('hidden');
        fromEl.classList.remove('opacity-0');
        
        toEl.classList.remove('hidden');
        toEl.classList.add('opacity-0');
        void toEl.offsetWidth; // force layout trigger
        toEl.classList.remove('opacity-0');
        toEl.classList.add('transition-opacity', 'duration-200');
      }, 200);
    }

    // Navigation back triggers
    function backToHome() {
      switchScreen('screen-purchase', 'screen-home');
    }

    function resetAndBackToHome() {
      clearInterval(timerInterval);
      clearTimeout(toastTimeout1);
      clearTimeout(toastTimeout2);
      
      currentSelection = {
        menuKey: null,
        size: null,
        qty: 1
      };
      
      switchScreen('screen-payment', 'screen-home');
    }
